99 void GestionContact::creerTodos(Interaction* i)
100 {
101  const string delimiter = "@todo ";
102  const int delimiterLength = delimiter.length();
103 
104  size_t position;
105  size_t last = 0;
106 
107  string resume = i->getResume();
108  position = resume.find(delimiter, last);
109  while(position != string::npos) // on recherche tous les @ todo
110  {
111  size_t positionEndLine = resume.find("\n", position); // fin de ligne du todo
112  if(positionEndLine == string::npos)
113  positionEndLine = resume.length(); // fin du todo a la fin du texte si pas de nouvelle ligne
114 
115  size_t positionNextTodo = resume.find(delimiter, position + delimiterLength); //@todo present sur la meme ligne
116  if(positionNextTodo != string::npos && positionNextTodo < positionEndLine)
117  positionEndLine = positionNextTodo;
118 
119  string todoText = resume.substr(position, positionEndLine - position); // on decoupe le texte correspondant au todo
120  Date d = getDateFromLine(todoText);
121  Todo* t = new Todo(todoText, d, false);
122  resume = resume.substr(0, position) + todoText + resume.substr(positionEndLine); // correction de la ligne du todo avec celle analysee et retournee par l'instance
123  i->ajoutTodo(t);
124 
125  last = position + delimiterLength;
126  position = resume.find(delimiter, last);
127  }
128 }

138 const Date GestionContact::getDateFromLine(const string& text)
139 {
140  Date d = Date();
141  const string delimiter = "@date ";
142  const int delimiterLength = delimiter.length();
143 
144  size_t position = text.find(delimiter);
145 
146  if(position != string::npos) // recherche de @ date, sinon date du jour
147  {
148  position += delimiterLength;
149  size_t last = position;
150  string dateText = text.substr(position);
151 
152  size_t positionFinDate = 0;
153  for(int i = 0; i < 2; i++) // position du 'J/M/' pour le debut de la date
154  {
155  positionFinDate = dateText.find("/", positionFinDate);
156  if(positionFinDate == string::npos)
157  {
158  throw invalid_argument("Date mal formee dans un todo");
159  }
160  positionFinDate++;
161  }
162  positionFinDate += 4; //YYYY
163  if(positionFinDate > dateText.length())
164  {
165  throw invalid_argument("Date mal formee dans un todo");
166  }
167  dateText = dateText.substr(0, positionFinDate); // on coupe la fin si @ date ne termine pas la ligne
168  d = Date(dateText);
169 
170  position = text.find(delimiter, last);
171  if(position != string::npos)
172  {
173  throw invalid_argument("Plusieurs @date dans une ligne de todo");
174  }
175  }
