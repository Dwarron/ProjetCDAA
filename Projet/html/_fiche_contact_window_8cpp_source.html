201 void FicheContactWindow::modifFiche()

203  string nom = ui->nomLineEdit->text().toStdString();
204  string nomTest = Verificator::onlyLetters(nom);
205 
206  //on va regarder quel champ sont modifies pour utiliser les setters/mutateur
207  if(curContact->getNom() != nom)
208  {
209  if(nom.length() == 0)
210  {
211  QMessageBox::warning(this, "Nom vide", QString::fromStdString("Veuillez entrer un nom"));
212  }
213  else if(nomTest != nom)
214  {
215  QMessageBox::StandardButton confirmation = QMessageBox::question(this, QString::fromStdString("Nom invalide : " + nom), QString::fromStdString("Vouliez-vous écrire : " + nomTest + " ?"));

217  {
218  ui->nomLineEdit->setText(QString::fromStdString(nomTest));

220  }

222  {

224  }
225  }
226 
227  string prenom = ui->prenomLineEdit->text().toStdString();
228  string prenomTest = Verificator::onlyLetters(prenom);
229 
230  if(curContact->getPrenom() != prenom)

232  if(prenom.length() == 0)
233  {
234  QMessageBox::warning(this, "Prenom vide", QString::fromStdString("Veuillez entrer un prenom"));

237  {
238  QMessageBox::StandardButton confirmation = QMessageBox::question(this, QString::fromStdString("Prenom invalide : " + prenom), QString::fromStdString("Vouliez-vous écrire : " + prenomTest + " ?"));
