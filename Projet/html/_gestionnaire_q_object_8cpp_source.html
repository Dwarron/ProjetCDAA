454 void GestionnaireQObject::exportJSON(QString filePath)
455 {
456  QFile file(filePath);
457  if(file.open(QIODevice::WriteOnly))
458  {
459  QJsonDocument doc;
460  QJsonArray baseArray;
461  QJsonArray contactsArray;
462  QJsonArray interactionsArray;
463  list<Interaction*> interactions = gestionContact->getAllInteractions();
464  map<Interaction*, int> interactionsIds;
465 
466  int id = 0;
467  for(auto it = interactions.begin(); it != interactions.end(); it++)
468  {
469  interactionsIds[*it] = id;
470  QJsonObject interactionJson = JSONSerializer::serialize(*it);
471  interactionJson.insert("id", id);
472  interactionsArray.append(interactionJson);
473  id++;
474  }
475 
476  for(auto c = gestionContact->getContacts().begin(); c != gestionContact->getContacts().end(); c++)
477  {
478  QJsonObject contactJson = JSONSerializer::serialize(*c);
479 
480  QJsonArray contactInteractionArray;
481  for(auto i = (*c)->getInteractions().begin(); i != (*c)->getInteractions().end(); i++)
482  {
483  contactInteractionArray.append(QJsonValue::fromVariant(QVariant(interactionsIds[*i])));
484  }
485 
486  contactJson.insert("interactionsIds", contactInteractionArray);
487  contactsArray.append(contactJson);
488  }
489 
490  QJsonArray infosArray;
491  QJsonObject infosObj;
492  infosObj.insert("dernierSuppression", QString::fromStdString(gestionContact->getDateDerniereSuppression().toString()));
493  infosArray.append(infosObj);
494 
495  baseArray.append(contactsArray);
496  baseArray.append(interactionsArray);
497  baseArray.append(infosArray);
498  doc.setArray(baseArray);
499 
500  file.write(doc.toJson());
501  file.close();
502  }
503  else
504  {
505  throw runtime_error("Unable to open JSON file");
506  }
507 }

516 void GestionnaireQObject::importJSON(QString filePath)
517 {
518  QFile file(filePath);
519  if (file.open(QIODevice::ReadOnly))
520  {
521  QByteArray bytes = file.readAll();
522  QJsonDocument doc = QJsonDocument::fromJson(bytes);
523  QJsonArray baseArray = doc.array();
524  QJsonArray contactsArray = baseArray.takeAt(0).toArray();
525  QJsonArray interactionsArray = baseArray.takeAt(0).toArray();
526  QJsonArray infosArray = baseArray.takeAt(0).toArray();
527 
528  emit onLoadJSON();
529 
530  for(auto it = contactsArray.begin(); it != contactsArray.end(); it++)
531  {
532  QJsonObject contactObj = (*it).toObject();
533  emit contactCreated(contactObj["nom"].toString().toStdString(), contactObj["prenom"].toString().toStdString(),
534  contactObj["entreprise"].toString().toStdString(), contactObj["tel"].toString().toStdString(),
535  contactObj["mail"].toString().toStdString(), contactObj["photo"].toString().toStdString(),
536  Date(contactObj["dateCreation"].toString().toStdString()), Date(contactObj["dateModification"].toString().toStdString()));
537  }
538 
539  map<int, list<Contact*>> interactionsToContacts;
540 
541  list<Contact*> contacts = gestionContact->getContacts();
542  auto cJson = contactsArray.begin();
543  for(auto c = contacts.begin(); c != contacts.end(); c++, cJson++)
544  {
545  QJsonArray interactionsContactToArray = (*cJson).toObject()["interactionsIds"].toArray();
546  for(auto i = interactionsContactToArray.begin(); i != interactionsContactToArray.end(); i++)
547  {
548  int interactionId = (*i).toInt();
549  interactionsToContacts[interactionId].push_back(*c);
550  }
551  }
552 
553  for(auto it = interactionsArray.begin(); it != interactionsArray.end(); it++)
554  {
555  QJsonObject interactionObj = (*it).toObject();
556  int indice = interactionObj["id"].toInt();
557  Interaction* i = ajoutInteraction(*interactionsToContacts[indice].begin(), Date(interactionObj["date"].toString().toStdString()),
558  interactionObj["resume"].toString(), false);
559 
560  QJsonArray todosArray = interactionObj["todos"].toArray();
561  for(auto t = todosArray.begin(); t != todosArray.end(); t++)
562  {
563  QJsonObject todoObj = (*t).toObject();
564  i->ajoutTodo(new Todo(todoObj["contenu"].toString().toStdString(), Date(todoObj["echeance"].toString().toStdString()),
565  todoObj["effectue"].toBool()));
566  }
567  emit onCreateTodos(i, i->getTodos());
568 
569  auto c = interactionsToContacts[indice].begin();
570  c++;
571  for(; c != interactionsToContacts[indice].end(); c++)
572  {
573  ajoutInteraction(*c, i);
574  }
575  }
576 
577  string date = infosArray.begin()->toObject()["dernierSuppression"].toString().toStdString();
578  gestionContact->setDateDerniereSuppression(Date(date));
579  emit contactDeleted(date);
580  }
581  else
582  {
583  throw runtime_error("Unable to open JSON file");
584  }
585 }
